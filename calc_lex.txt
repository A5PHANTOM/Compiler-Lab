%{
#include <stdio.h>
#include <stdlib.h>

int op = 0, i;
float a, b;

void digi(void);
%}

dig     [0-9]+|([0-9]*)"."([0-9]+)
add     \+
sub     \-
mul     \*
div     \/
pow     \^
ln      \n

%%
{dig}   { digi(); }
{add}   { op = 1; }
{sub}   { op = 2; }
{mul}   { op = 3; }
{div}   { op = 4; }
{pow}   { op = 5; }
{ln}    { printf("\nThe Answer: %f\n\n", a); }
[ \t]+  ;   /* ignore spaces */
%%

void digi(void)
{
    if (op == 0)
        a = atof(yytext);
    else {
        b = atof(yytext);
        switch (op)
        {
            case 1: a = a + b; break;
            case 2: a = a - b; break;
            case 3: a = a * b; break;
            case 4: a = a / b; break;
            case 5: {
                float result = 1;
                for (i = 0; i < (int)b; i++)
                    result *= a;
                a = result;
                break;
            }
        }
        op = 0;
    }
}

int main(void)
{
    printf("Enter expression (e.g.,2+3): ");
    yylex();
    return 0;
}
