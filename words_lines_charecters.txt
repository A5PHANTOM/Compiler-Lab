Program(lex)
 
%{
#include<stdio.h>
int line=0,word=0,ch=0;
%}
%%
[a-zA-Z0-9]+ {word++;ch+= yyleng;}
[\t] {ch +=yyleng;}
\n {line++; ch++;}
. {ch++;}
%%
main(int argc,char*argv[])
{
yyin=fopen(argv[1],"r");
yylex();
printf("line=%d\n",line);
printf("word=%d\n",word);
printf("character=%d\n",ch);
}
yywrap()
{
return 1;
}
 
 
 
 
 
Output
hello world
hi lex
line=2
word=4
character=19
